                <input type="radio" id="domicilio" name="tipo_entrega" value="DOMICILIO" onchange="mostrarOcultarDireccion()">

    <script>
        
        // ====================================================================
        // COPIA DEL CÓDIGO server.js PARA DEMOSTRACIÓN (En producción, se importarían)
        // La lógica de cálculo se mantiene sin cambios, asegurando la consistencia.
        // ====================================================================
        const COMISION_CLIENTE = 0.02; 
        const TARIFA_DIURNA = 25.00; 
        const TARIFA_NOCTURNA = 35.00; 
        const GANANCIA_MOTO_AUTO = 0.75; 
        const GANANCIA_BICI_TRICICLO = 0.65; 
        const GANANCIA_SCOOTER_CAMINANDO = 0.60; 
        const LIMITE_BICI = 2.5; 
        const LIMITE_SCOOTER = 1.5; 
        
        function obtenerCuotaFija(horaPedido) {
            const hora = horaPedido.getHours();
            if (hora >= 9 && hora < 21) {
                return TARIFA_DIURNA; 
            } else {
                return TARIFA_NOCTURNA; 
            }
        }

        // Función original de server.js (simplificada para solo calcular los valores del cliente)
        function calcularCostoYComisionesCliente(subtotalCompra, tipoEntrega, tipoVehiculoAsignado = null, distanciaKm = null) {
            const horaPedido = new Date(); 
            let costoEnvio = 0.00;
            
            if (typeof subtotalCompra !== 'number' || subtotalCompra <= 0) {
                throw new Error("El subtotal de la compra debe ser un número positivo.");
            }

            const comisionPlataformaCliente = subtotalCompra * COMISION_CLIENTE;
            let tarifaAplicada = 'N/A';
            
            if (tipoEntrega === 'DOMICILIO') {
                if (!tipoVehiculoAsignado || typeof distanciaKm !== 'number') {
                     throw new Error("Para Envío a Domicilio, se requiere el tipo de vehículo y la distancia en km.");
                }

                const cuotaFija = obtenerCuotaFija(horaPedido);
                costoEnvio = cuotaFija;
                tarifaAplicada = costoEnvio === TARIFA_DIURNA ? "DIURNA ($25)" : "NOCTURNA ($35)";
                
                let limiteMaximo = Infinity;

                switch (tipoVehiculoAsignado) {
                    case 'MOTO':
                    case 'AUTO':
                        break;
                    case 'BICI':
                    case 'TRICICLO':
                        limiteMaximo = LIMITE_BICI;
                        break;
                    case 'SCOOTER':
                    case 'CAMINANDO':
                        limiteMaximo = LIMITE_SCOOTER;
                        break;
                    default:
                        throw new Error("Medio de entrega no válido o no asignado.");
                }
                
                if (distanciaKm > limiteMaximo) {
                     throw new Error(`🚫 ERROR: El vehículo (${tipoVehiculoAsignado}) solo puede cubrir hasta ${limiteMaximo.toFixed(1)} km. Tu entrega requiere ${distanciaKm.toFixed(2)} km.`);
                }
                
            } else if (tipoEntrega !== 'RECOLECTA') {
                throw new Error("Tipo de entrega no válido.");
            }
            
            const totalCliente = subtotalCompra + comisionPlataformaCliente + costoEnvio;

            return {
                subtotalCompra: subtotalCompra.toFixed(2),
                comisionCliente: comisionPlataformaCliente.toFixed(2),
                costoEnvio: costoEnvio.toFixed(2),
                totalCliente: totalCliente.toFixed(2),
                tarifaAplicada: tarifaAplicada
            };
        }
        // ====================================================================

        const form = document.getElementById('checkout-form');
        const resultadoDiv = document.getElementById('resultado-calculo');
        const direccionGroup = document.getElementById('direccion-group');
        const alertaError = document.getElementById('alerta-error');

        function mostrarOcultarDireccion() {
            if (document.getElementById('domicilio').checked) {
                direccionGroup.style.display = 'block';
                document.getElementById('direccion').setAttribute('required', 'required');
                document.getElementById('distancia').setAttribute('required', 'required');
            } else {
                direccionGroup.style.display = 'none';
                document.getElementById('direccion').removeAttribute('required');
                document.getElementById('distancia').removeAttribute('required');
                alertaError.style.display = 'none';
                resultadoDiv.style.display = 'none';
            }
        }

        form.addEventListener('submit', function(event) {
            event.preventDefault();
            alertaError.style.display = 'none';
            resultadoDiv.style.display = 'none';
            
            const subtotal = parseFloat(document.getElementById('subtotal').value);
            const tipoEntrega = document.querySelector('input[name="tipo_entrega"]:checked').value;
            
            let distancia = null;
            let vehiculo = null;

            if (tipoEntrega === 'DOMICILIO') {
                distancia = parseFloat(document.getElementById('distancia').value);
                vehiculo = document.getElementById('vehiculo').value;
                if (isNaN(distancia) || distancia <= 0 || !document.getElementById('direccion').value) {
                    mostrarError("Por favor, ingresa una distancia y dirección válidas.");
                    return;
                }
            }
            
            try {
                // Llamada a la función del "back-end"
                const resultados = calcularCostoYComisionesCliente(subtotal, tipoEntrega, vehiculo, distancia);
                
                // Mostrar resultados
                document.getElementById('res-subtotal').textContent = `$${resultados.subtotalCompra} MXN`;
                document.getElementById('res-comision-cliente').textContent = `$${resultados.comisionCliente} MXN`;
                document.getElementById('res-costo-envio').textContent = `$${resultados.costoEnvio} MXN`;
                document.getElementById('res-total').textContent = `$${resultados.totalCliente} MXN`;
                document.getElementById('res-tarifa-aplicada').textContent = resultados.tarifaAplicada;

                resultadoDiv.style.display = 'block';

                // NOTA: Aquí se realizaría la integración con la pasarela de pagos (Mercado Pago, PayPal, etc.)
                // Después del pago exitoso, la información se enviaría a tu servidor para la asignación del repartidor.

            } catch (error) {
                mostrarError(error.message);
            }
        });

        function mostrarError(mensaje) {
            alertaError.textContent = mensaje;
            alertaError.style.display = 'block';
        }

    </script>
